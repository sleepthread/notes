/* 
冒泡排序是排序数组的一种算法. 复杂度为 O(N^2)
*/
/*
*  使用双层循环
*/
function bubbleSort (items) {
  const length = items.length

  for (let i = (length - 1); i > 0; i--) {
    for (let j = (length - i); j > 0; j--) {
      if (items[j] < items[j - 1]) {
        [items[j], items[j - 1]] = [items[j - 1], items[j]]
      }
    }
  }

  return items
}
